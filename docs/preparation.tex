\paragraph{Die Elektrische Spannung:}
\begin{center}
	\begin{tabular}{ |c|c| }
		Formelzeichen & U            \\
		Einheit       & Volt (V)     \\
		Beispiel      & $ U = 230V $ \\
	\end{tabular}
\end{center}

\paragraph{Die Elektrische Stromstärke:}
\begin{center}
	\begin{tabular}{ |c|c| }
		Formelzeichen & I            \\
		Einheit       & Ambere (A)   \\
		Beispiel      & $ I = 2A $   \\
	\end{tabular}
\end{center}

\paragraph{Der Elektrische Widerstand:}
\begin{center}
	\begin{tabular}{ |c|c| }
		Formelzeichen & R            \\
		Einheit       & Ohm (\Omega)   \\
		Beispiel      & $ R = 10\Omega $   \\
	\end{tabular}
\end{center}

\paragraph{Das Ohmsche Gesetz:}
\begin{center}
    \begin{align}
    	U = R \cdot I
    	P = U \cdot I
    \end{align}
\end{align}

\paragraph{Mathematische Zusammenhänge der Reihenschaltung:}
Es gilt: 
\begin{center}
	\begin{align}
		U_G = U_1 + U_2 + \dots + U_n
		I_G = I_1 = I_2 = I_3 = \dots = I_n
		R_G = R_1 + R_2 + \dots + R_n
	\end{align}
\begin{center}
